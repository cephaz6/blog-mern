# # Use Node.js as base image for both frontend and backend
# FROM node:21.2.0-alpine

# # Set working directory for frontend
# WORKDIR /usr/src/frontend

# # Copy and install frontend dependencies
# COPY ./package*.json ./

# RUN npm install

# # Copy frontend files
# COPY  . .

# # Build frontend
# RUN npm run build

# # Backend
# FROM node:21.2.0-alpine

# # Set working directory for backend
# WORKDIR /usr/src/backend

# # Copy and install backend dependencies
# COPY  ./package*.json ./

# RUN npm install

# # Copy backend files
# COPY   . .

# # Expose backend port
# EXPOSE 9000

# # Start the Node.js server (assuming server.js is your entry point)
# CMD ["node", "server.js"]
